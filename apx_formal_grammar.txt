var_list:
	tk_VAR var_type ident_list
	|
	tk_VAR var_type ident_list tk_ASSIGN value_list

var_type: 
	user_type
	|
	user_type tk_MUL
	|
	builtin_type
	|
	builtin_type tk_MUL

user_type:
	tk_IDENT
	|
	qualified_user_type

qualified_user_type:
	tk_IDENT tk_DOT tk_IDENT

builtin_type:
	tk_T_INT
	|
        tk_T_BOOL 
	.... 

ident_list:
	tk_IDENT
	|
	tk_COMMA ident_list;

value_list:
	assing
	|
	tk_COMMA assign_list

	tk_IDENT
        |
	tk_STRINGLIT
	|
